 16def _find_best_configuration():
 17    if torch.cuda.is_available():
 18
 19        # Check how much memory we have and select the best matching GPU
 20        # for the available VRAM size.
 21        _, vram = torch.cuda.mem_get_info()
 22        vram = vram / 1e9  # in GB
 23
 24        # Maybe we can get more configurations in the future.
 25        if vram > 80:  # More than 80 GB: use the A100 configurations.
 26            return "A100"
 27        elif vram > 30:  # More than 30 GB: use the V100 configurations.
 28            return "V100"
 29        elif vram > 14:  # More than 14 GB: use the RTX5000 configurations.
 30            return "rtx5000"
 31        else:  # Otherwise: not enough memory to train on the GPU, use CPU instead.
 32            return "CPU"
 33    else:
 34        return "CPU"

165    def _get_loaders(self):
166        batch_size = 1
167        num_workers = 1 if str(self.device) == "cpu" else 4
168
169        patch_shape = (self.patch_x, self.patch_y)
170        dataset = default_sam_dataset(
171            str(self.raw_path), self.raw_key, str(self.label_path), self.label_key,
172            patch_shape=patch_shape, with_segmentation_decoder=self.with_segmentation_decoder,
173        )
174
175        raw_path_val, label_path_val = self.raw_path_val, self.label_path_val
176        if raw_path_val is None:
177            # Use 10% of the dataset - at least one image - for validation.
178            n_val = min(1, int(0.1 * len(dataset)))
179            train_dataset, val_dataset = random_split(dataset, lengths=[len(dataset) - n_val, n_val])
180        else:
181            train_dataset = dataset
182            val_dataset = default_sam_dataset(
183                str(raw_path_val), self.raw_key, str(label_path_val), self.label_key,
184                patch_shape=patch_shape, with_segmentation_decoder=self.with_segmentation_decoder,
185            )
186
187        train_loader = torch_em.segmentation.get_data_loader(
188            train_dataset, batch_size=batch_size, shuffle=True, num_workers=num_workers,
189        )
190        val_loader = torch_em.segmentation.get_data_loader(
191            val_dataset, batch_size=batch_size, shuffle=True, num_workers=num_workers,
192        )
193        return train_loader, val_loader

195    def _get_model_type(self):
196        # Consolidate initial model name, the checkpoint path and the model type according to the configuration.
197        if self.initial_model is None or self.initial_model in ("None", ""):
198            model_type = CONFIGURATIONS[self.configuration]["model_type"]
199        else:
200            model_type = self.initial_model[:5]
201            if model_type != CONFIGURATIONS[self.configuration]["model_type"]:
202                warnings.warn(
203                    f"You have changed the model type for your chosen configuration {self.configuration} "
204                    f"from {CONFIGURATIONS[self.configuration]['model_type']} to {model_type}. "
205                    "The training may be very slow or not work at all."
206                )
207        assert model_type is not None
208        return model_type

213    def _validate_inputs(self):
214        missing_raw = self.raw_path is None or not os.path.exists(self.raw_path)
215        missing_label = self.label_path is None or not os.path.exists(self.label_path)
216        if missing_raw or missing_label:
217            msg = ""
218            if missing_raw:
219                msg += "The path to raw data is missing or does not exist. "
220            if missing_label:
221                msg += "The path to label data is missing or does not exist."
222            return widgets._generate_message("error", msg)
223        return False

225    def __call__(self, skip_validate=False):
226        if not skip_validate and self._validate_inputs():
227            return
228
229        # Set up progress bar and signals for using it within a threadworker.
230        pbar, pbar_signals = widgets._create_pbar_for_threadworker()
231
232        model_type = self._get_model_type()
233        if self.checkpoint is None:
234            model_registry = util.models()
235            checkpoint_path = model_registry.fetch(model_type)
236        else:
237            checkpoint_path = self.checkpoint
238
239        @thread_worker()
240        def run_training():
241            train_loader, val_loader = self._get_loaders()
242            train_sam_for_configuration(
243                name=self.name, configuration=self.configuration,
244                train_loader=train_loader, val_loader=val_loader,
245                checkpoint_path=checkpoint_path,
246                with_segmentation_decoder=self.with_segmentation_decoder,
247                model_type=model_type, device=self.device,
248                n_epochs=self.n_epochs, pbar_signals=pbar_signals,
249            )
250
251            # The best checkpoint after training.
252            export_checkpoint = os.path.join("checkpoints", self.name, "best.pt")
253            assert os.path.exists(export_checkpoint), export_checkpoint
254
255            # Export the model if an output path was given.
256            if self.output_path:
257
258                # If the output path has a pytorch specific ending then
259                # we just export the checkpoint.
260                if os.path.splitext(self.output_path)[1] in (".pt", ".pth"):
261                    util.export_custom_sam_model(
262                        checkpoint_path=export_checkpoint, model_type=model_type, save_path=self.output_path,
263                    )
264
265                # Otherwise we export it as bioimage.io model.
266                else:
267                    from micro_sam.bioimageio import export_sam_model
268
269                    # Load image and label image from the val loader.
270                    with torch.no_grad():
271                        image, label_image = next(iter(val_loader))
272                        image, label_image = image.cpu().numpy().squeeze(), label_image.cpu().numpy().squeeze()
273
274                    # Select the last channel of the label image if we have a channel axis.
275                    # (This contains the labels.)
276                    if label_image.ndim == 3:
277                        label_image = label_image[0]
278                    assert image.shape == label_image.shape
279                    label_image = label_image.astype("uint32")
280
281                    export_sam_model(
282                        image=image,
283                        label_image=label_image,
284                        model_type=model_type,
285                        name=self.name,
286                        output_path=self.output_path,
287                        checkpoint_path=export_checkpoint,
288                    )
289
290                pbar_signals.pbar_stop.emit()
291                return self.output_path
292
293            else:
294                pbar_signals.pbar_stop.emit()
295                return export_checkpoint
296
297        worker = run_training()
298        worker.returned.connect(lambda path: print(f"Training has finished. The trained model is saved at {path}."))
299        worker.start()
300        return worker
