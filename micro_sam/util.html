148def _get_default_device():
149    # check that we're in CI and use the CPU if we are
150    # otherwise the tests may run out of memory on MAC if MPS is used.
151    if os.getenv("GITHUB_ACTIONS") == "true":
152        return "cpu"
153    # Use cuda enabled gpu if it's available.
154    if torch.cuda.is_available():
155        device = "cuda"
156    # As second priority use mps.
157    # See https://pytorch.org/docs/stable/notes/mps.html for details
158    elif torch.backends.mps.is_available() and torch.backends.mps.is_built():
159        print("Using apple MPS device.")
160        device = "mps"
161    # Use the CPU as fallback.
162    else:
163        device = "cpu"
164    return device

167def get_device(device: Optional[Union[str, torch.device]] = None) -> Union[str, torch.device]:
168    """Get the torch device.
169
170    If no device is passed the default device for your system is used.
171    Else it will be checked if the device you have passed is supported.
172
173    Args:
174        device: The input device.
175
176    Returns:
177        The device.
178    """
179    if device is None or device == "auto":
180        device = _get_default_device()
181    else:
182        device_type = device if isinstance(device, str) else device.type
183        if device_type.lower() == "cuda":
184            if not torch.cuda.is_available():
185                raise RuntimeError("PyTorch CUDA backend is not available.")
186        elif device_type.lower() == "mps":
187            if not (torch.backends.mps.is_available() and torch.backends.mps.is_built()):
188                raise RuntimeError("PyTorch MPS backend is not available or is not built correctly.")
189        elif device_type.lower() == "cpu":
190            pass  # cpu is always available
191        else:
192            raise RuntimeError(f"Unsupported device: {device}\n"
193                               "Please choose from 'cpu', 'cuda', or 'mps'.")
194    return device

209def get_sam_model(
210    model_type: str = _DEFAULT_MODEL,
211    device: Optional[Union[str, torch.device]] = None,
212    checkpoint_path: Optional[Union[str, os.PathLike]] = None,
213    return_sam: bool = False,
214) -> SamPredictor:
215    r"""Get the SegmentAnything Predictor.
216
217    This function will download the required model or load it from the cached weight file.
218    This location of the cache can be changed by setting the environment variable: MICROSAM_CACHEDIR.
219    The name of the requested model can be set via `model_type`.
220    See https://computational-cell-analytics.github.io/micro-sam/micro_sam.html#finetuned-models
221    for an overview of the available models
222
223    Alternatively this function can also load a model from weights stored in a local filepath.
224    The corresponding file path is given via `checkpoint_path`. In this case `model_type`
225    must be given as the matching encoder architecture, e.g. "vit_b" if the weights are for
226    a SAM model with vit_b encoder.
227
228    By default the models are downloaded to a folder named 'micro_sam/models'
229    inside your default cache directory, eg:
230    * Mac: ~/Library/Caches/<AppName>
231    * Unix: ~/.cache/<AppName> or the value of the XDG_CACHE_HOME environment variable, if defined.
232    * Windows: C:\Users\<user>\AppData\Local\<AppAuthor>\<AppName>\Cache
233    See the pooch.os_cache() documentation for more details:
234    https://www.fatiando.org/pooch/latest/api/generated/pooch.os_cache.html
235
236    Args:
237        model_type: The SegmentAnything model to use. Will use the standard vit_h model by default.
238            To get a list of all available model names you can call `get_model_names`.
239        device: The device for the model. If none is given will use GPU if available.
240        checkpoint_path: The path to a file with weights that should be used instead of using the
241            weights corresponding to `model_type`. If given, `model_type` must match the architecture
242            corresponding to the weight file. E.g. if you use weights for SAM with vit_b encoder
243            then `model_type` must be given as "vit_b".
244        return_sam: Return the sam model object as well as the predictor.
245
246    Returns:
247        The segment anything predictor.
248    """
249    device = get_device(device)
250
251    # We support passing a local filepath to a checkpoint.
252    # In this case we do not download any weights but just use the local weight file,
253    # as it is, without copying it over anywhere or checking it's hashes.
254
255    # checkpoint_path has not been passed, we download a known model and derive the correct
256    # URL from the model_type. If the model_type is invalid pooch will raise an error.
257    if checkpoint_path is None:
258        model_registry = models()
259        checkpoint = model_registry.fetch(model_type)
260    # checkpoint_path has been passed, we use it instead of downloading a model.
261    else:
262        # Check if the file exists and raise an error otherwise.
263        # We can't check any hashes here, and we don't check if the file is actually a valid weight file.
264        # (If it isn't the model creation will fail below.)
265        if not os.path.exists(checkpoint_path):
266            raise ValueError(f"Checkpoint at {checkpoint_path} could not be found.")
267        checkpoint = checkpoint_path
268
269    # Our fine-tuned model types have a suffix "_...". This suffix needs to be stripped
270    # before calling sam_model_registry.
271    abbreviated_model_type = model_type[:5]
272    if abbreviated_model_type not in _MODEL_TYPES:
273        raise ValueError(f"Invalid model_type: {abbreviated_model_type}. Expect one of {_MODEL_TYPES}")
274    if abbreviated_model_type == "vit_t" and not VIT_T_SUPPORT:
275        raise RuntimeError(
276            "mobile_sam is required for the vit-tiny."
277            "You can install it via 'pip install git+https://github.com/ChaoningZhang/MobileSAM.git'"
278        )
279
280    sam = sam_model_registry[abbreviated_model_type](checkpoint=checkpoint)
281    sam.to(device=device)
282    predictor = SamPredictor(sam)
283    predictor.model_type = abbreviated_model_type
284    if return_sam:
285        return predictor, sam
286    return predictor

289# We write a custom unpickler that skips objects that cannot be found instead of
290# throwing an AttributeError or ModueNotFoundError.
291# NOTE: since we just want to unpickle the model to load its weights these errors don't matter.
292# See also https://stackoverflow.com/questions/27732354/unable-to-load-files-using-pickle-and-multiple-modules
293class _CustomUnpickler(pickle.Unpickler):
294    def find_class(self, module, name):
295        try:
296            return super().find_class(module, name)
297        except (AttributeError, ModuleNotFoundError) as e:
298            warnings.warn(f"Did not find {module}:{name} and will skip it, due to error {e}")
299            return None

302def get_custom_sam_model(
303    checkpoint_path: Union[str, os.PathLike],
304    model_type: str = "vit_h",
305    device: Optional[Union[str, torch.device]] = None,
306    return_sam: bool = False,
307    return_state: bool = False,
308) -> SamPredictor:
309    """Load a SAM model from a torch_em checkpoint.
310
311    This function enables loading from the checkpoints saved by
312    the functionality in `micro_sam.training`.
313
314    Args:
315        checkpoint_path: The path to the corresponding checkpoint if not in the default model folder.
316        model_type: The SegmentAnything model_type for the given checkpoint.
317        device: The device for the model. If none is given will use GPU if available.
318        return_sam: Return the sam model object as well as the predictor.
319        return_state: Return the full state of the checkpoint in addition to the predictor.
320
321    Returns:
322        The segment anything predictor.
323    """
324    assert not (return_sam and return_state)
325
326    # over-ride the unpickler with our custom one
327    custom_pickle = pickle
328    custom_pickle.Unpickler = _CustomUnpickler
329
330    device = get_device(device)
331    sam = sam_model_registry[model_type]()
332
333    # load the model state, ignoring any attributes that can't be found by pickle
334    state = torch.load(checkpoint_path, map_location=device, pickle_module=custom_pickle)
335    model_state = state["model_state"]
336
337    # copy the model weights from torch_em's training format
338    sam_prefix = "sam."
339    model_state = OrderedDict(
340        [(k[len(sam_prefix):] if k.startswith(sam_prefix) else k, v) for k, v in model_state.items()]
341    )
342    sam.load_state_dict(model_state)
343    sam.to(device)
344
345    predictor = SamPredictor(sam)
346    predictor.model_type = model_type
347
348    if return_sam:
349        return predictor, sam
350    if return_state:
351        return predictor, state
352    return predictor

355def export_custom_sam_model(
356    checkpoint_path: Union[str, os.PathLike],
357    model_type: str,
358    save_path: Union[str, os.PathLike],
359) -> None:
360    """Export a finetuned segment anything model to the standard model format.
361
362    The exported model can be used by the interactive annotation tools in `micro_sam.annotator`.
363
364    Args:
365        checkpoint_path: The path to the corresponding checkpoint if not in the default model folder.
366        model_type: The SegmentAnything model type corresponding to the checkpoint (vit_h, vit_b, vit_l or vit_t).
367        save_path: Where to save the exported model.
368    """
369    _, state = get_custom_sam_model(
370        checkpoint_path, model_type=model_type, return_state=True, device="cpu",
371    )
372    model_state = state["model_state"]
373    prefix = "sam."
374    model_state = OrderedDict(
375        [(k[len(prefix):] if k.startswith(prefix) else k, v) for k, v in model_state.items()]
376    )
377    torch.save(model_state, save_path)

391def _to_image(input_):
392    # we require the input to be uint8
393    if input_.dtype != np.dtype("uint8"):
394        # first normalize the input to [0, 1]
395        input_ = input_.astype("float32") - input_.min()
396        input_ = input_ / input_.max()
397        # then bring to [0, 255] and cast to uint8
398        input_ = (input_ * 255).astype("uint8")
399    if input_.ndim == 2:
400        image = np.concatenate([input_[..., None]] * 3, axis=-1)
401    elif input_.ndim == 3 and input_.shape[-1] == 3:
402        image = input_
403    else:
404        raise ValueError(f"Invalid input image of shape {input_.shape}. Expect either 2D grayscale or 3D RGB image.")
405    return image

408def _precompute_tiled_2d(predictor, input_, tile_shape, halo, f, verbose=True):
409    tiling = blocking([0, 0], input_.shape[:2], tile_shape)
410    n_tiles = tiling.numberOfBlocks
411
412    f.attrs["input_size"] = None
413    f.attrs["original_size"] = None
414
415    features = f.require_group("features")
416    features.attrs["shape"] = input_.shape[:2]
417    features.attrs["tile_shape"] = tile_shape
418    features.attrs["halo"] = halo
419
420    for tile_id in tqdm(range(n_tiles), total=n_tiles, desc="Predict image embeddings for tiles", disable=not verbose):
421        tile = tiling.getBlockWithHalo(tile_id, list(halo))
422        outer_tile = tuple(slice(beg, end) for beg, end in zip(tile.outerBlock.begin, tile.outerBlock.end))
423
424        predictor.reset_image()
425        tile_input = _to_image(input_[outer_tile])
426        predictor.set_image(tile_input)
427        tile_features = predictor.get_image_embedding()
428        original_size = predictor.original_size
429        input_size = predictor.input_size
430
431        ds = features.create_dataset(
432            str(tile_id), data=tile_features.cpu().numpy(), compression="gzip", chunks=tile_features.shape
433        )
434        ds.attrs["original_size"] = original_size
435        ds.attrs["input_size"] = input_size
436
437    return features

440def _precompute_tiled_3d(predictor, input_, tile_shape, halo, f, verbose=True):
441    assert input_.ndim == 3
442
443    shape = input_.shape[1:]
444    tiling = blocking([0, 0], shape, tile_shape)
445    n_tiles = tiling.numberOfBlocks
446
447    f.attrs["input_size"] = None
448    f.attrs["original_size"] = None
449
450    features = f.require_group("features")
451    features.attrs["shape"] = shape
452    features.attrs["tile_shape"] = tile_shape
453    features.attrs["halo"] = halo
454
455    n_slices = input_.shape[0]
456    pbar = tqdm(total=n_tiles * n_slices, desc="Predict image embeddings for tiles and slices", disable=not verbose)
457
458    for tile_id in range(n_tiles):
459        tile = tiling.getBlockWithHalo(tile_id, list(halo))
460        outer_tile = tuple(slice(beg, end) for beg, end in zip(tile.outerBlock.begin, tile.outerBlock.end))
461
462        ds = None
463        for z in range(n_slices):
464            predictor.reset_image()
465            tile_input = _to_image(input_[z][outer_tile])
466            predictor.set_image(tile_input)
467            tile_features = predictor.get_image_embedding()
468
469            if ds is None:
470                shape = (input_.shape[0],) + tile_features.shape
471                chunks = (1,) + tile_features.shape
472                ds = features.create_dataset(
473                    str(tile_id), shape=shape, dtype="float32", compression="gzip", chunks=chunks
474                )
475
476            ds[z] = tile_features.cpu().numpy()
477            pbar.update(1)
478
479        original_size = predictor.original_size
480        input_size = predictor.input_size
481
482        ds.attrs["original_size"] = original_size
483        ds.attrs["input_size"] = input_size
484
485    return features

500def _precompute_2d(input_, predictor, save_path, tile_shape, halo):
501    f = zarr.open(save_path, "a")
502
503    use_tiled_prediction = tile_shape is not None
504    if "input_size" in f.attrs:  # the embeddings have already been precomputed
505        features = f["features"][:] if tile_shape is None else f["features"]
506        original_size, input_size = f.attrs["original_size"], f.attrs["input_size"]
507
508    elif use_tiled_prediction:  # the embeddings have not been computed yet and we use tiled prediction
509        features = _precompute_tiled_2d(predictor, input_, tile_shape, halo, f)
510        original_size, input_size = None, None
511
512    else:  # the embeddings have not been computed yet and we use normal prediction
513        image = _to_image(input_)
514        predictor.set_image(image)
515        features = predictor.get_image_embedding()
516        original_size, input_size = predictor.original_size, predictor.input_size
517        f.create_dataset("features", data=features.cpu().numpy(), chunks=features.shape)
518        f.attrs["input_size"] = input_size
519        f.attrs["original_size"] = original_size
520
521    image_embeddings = {
522        "features": features, "input_size": input_size, "original_size": original_size,
523    }
524    return image_embeddings

527def _compute_3d(input_, predictor):
528    features = []
529    original_size, input_size = None, None
530
531    for z_slice in tqdm(input_, desc="Precompute Image Embeddings"):
532        predictor.reset_image()
533
534        image = _to_image(z_slice)
535        predictor.set_image(image)
536        embedding = predictor.get_image_embedding()
537        features.append(embedding[None])
538
539        if original_size is None:
540            original_size = predictor.original_size
541        if input_size is None:
542            input_size = predictor.input_size
543
544    # concatenate across the z axis
545    features = torch.cat(features)
546
547    image_embeddings = {
548        "features": features.cpu().numpy(), "input_size": input_size, "original_size": original_size,
549    }
550    return image_embeddings

553def _precompute_3d(input_, predictor, save_path, lazy_loading, tile_shape=None, halo=None):
554    f = zarr.open(save_path, "a")
555
556    use_tiled_prediction = tile_shape is not None
557    if "input_size" in f.attrs:  # the embeddings have already been precomputed
558        features = f["features"]
559        original_size, input_size = f.attrs["original_size"], f.attrs["input_size"]
560
561    elif use_tiled_prediction:  # the embeddings have not been computed yet and we use tiled prediction
562        features = _precompute_tiled_3d(predictor, input_, tile_shape, halo, f)
563        original_size, input_size = None, None
564
565    else:  # the embeddings have not been computed yet and we use normal prediction
566        features = f["features"] if "features" in f else None
567        original_size, input_size = None, None
568
569        for z, z_slice in tqdm(enumerate(input_), total=input_.shape[0], desc="Precompute Image Embeddings"):
570            if features is not None:
571                emb = features[z]
572                if np.count_nonzero(emb) != 0:
573                    continue
574
575            predictor.reset_image()
576            image = _to_image(z_slice)
577            predictor.set_image(image)
578            embedding = predictor.get_image_embedding()
579
580            original_size, input_size = predictor.original_size, predictor.input_size
581            if features is None:
582                shape = (input_.shape[0],) + embedding.shape
583                chunks = (1,) + embedding.shape
584                features = f.create_dataset("features", shape=shape, chunks=chunks, dtype="float32")
585            features[z] = embedding.cpu().numpy()
586
587        f.attrs["input_size"] = input_size
588        f.attrs["original_size"] = original_size
589
590    # we load the data into memory if lazy loading was not specified
591    # and if we do not use tiled prediction (we cannot load the full tiled data structure into memory)
592    if not lazy_loading and not use_tiled_prediction:
593        features = features[:]
594
595    image_embeddings = {
596        "features": features, "input_size": input_size, "original_size": original_size,
597    }
598    return image_embeddings

606def precompute_image_embeddings(
607    predictor: SamPredictor,
608    input_: np.ndarray,
609    save_path: Optional[str] = None,
610    lazy_loading: bool = False,
611    ndim: Optional[int] = None,
612    tile_shape: Optional[Tuple[int, int]] = None,
613    halo: Optional[Tuple[int, int]] = None,
614    wrong_file_callback: Optional[Callable] = None,
615) -> ImageEmbeddings:
616    """Compute the image embeddings (output of the encoder) for the input.
617
618    If 'save_path' is given the embeddings will be loaded/saved in a zarr container.
619
620    Args:
621        predictor: The SegmentAnything predictor.
622        input_: The input data. Can be 2 or 3 dimensional, corresponding to an image, volume or timeseries.
623        save_path: Path to save the embeddings in a zarr container.
624        lazy_loading: Whether to load all embeddings into memory or return an
625            object to load them on demand when required. This only has an effect if 'save_path' is given
626            and if the input is 3 dimensional.
627        ndim: The dimensionality of the data. If not given will be deduced from the input data.
628        tile_shape: Shape of tiles for tiled prediction. By default prediction is run without tiling.
629        halo: Overlap of the tiles for tiled prediction.
630        wrong_file_callback [callable]: Function to call when an embedding file with wrong file signature
631            is passed. If none is given a wrong file signature will cause a warning.
632            The callback ,ust have the signature 'def callback(save_path: str) -> str',
633            where the return value is the (potentially updated) embedding save path.
634    """
635    ndim = input_.ndim if ndim is None else ndim
636    if tile_shape is not None:
637        assert save_path is not None, "Tiled prediction is only supported when the embeddings are saved to file."
638
639    if save_path is not None:
640        save_path = str(save_path)
641        data_signature = _compute_data_signature(input_)
642
643        f = zarr.open(save_path, "a")
644        key_vals = [
645            ("data_signature", data_signature),
646            ("tile_shape", tile_shape if tile_shape is None else list(tile_shape)),
647            ("halo", halo if halo is None else list(halo)),
648            ("model_type", predictor.model_type)
649        ]
650        if "input_size" in f.attrs:  # we have computed the embeddings already and perform checks
651            for key, val in key_vals:
652                if val is None:
653                    continue
654                # check whether the key signature does not match or is not in the file
655                if key not in f.attrs or f.attrs[key] != val:
656                    raise RuntimeError(
657                        f"Embeddings file {save_path} is invalid due to unmatching {key}: "
658                        f"{f.attrs.get(key)} != {val}.Please recompute embeddings in a new file."
659                    )
660                    if wrong_file_callback is not None:
661                        save_path = wrong_file_callback(save_path)
662                        f = zarr.open(save_path, "a")
663                    break
664
665        for key, val in key_vals:
666            if key not in f.attrs:
667                f.attrs[key] = val
668
669    if ndim == 2:
670        image_embeddings = _compute_2d(input_, predictor) if save_path is None else\
671            _precompute_2d(input_, predictor, save_path, tile_shape, halo)
672
673    elif ndim == 3:
674        image_embeddings = _compute_3d(input_, predictor) if save_path is None else\
675            _precompute_3d(input_, predictor, save_path, lazy_loading, tile_shape, halo)
676
677    else:
678        raise ValueError(f"Invalid dimesionality {input_.ndim}, expect 2 or 3 dim data.")
679
680    return image_embeddings

723def compute_iou(mask1: np.ndarray, mask2: np.ndarray) -> float:
724    """Compute the intersection over union of two masks.
725
726    Args:
727        mask1: The first mask.
728        mask2: The second mask.
729
730    Returns:
731        The intersection over union of the two masks.
732    """
733    overlap = np.logical_and(mask1 == 1, mask2 == 1).sum()
734    union = np.logical_or(mask1 == 1, mask2 == 1).sum()
735    eps = 1e-7
736    iou = float(overlap) / (float(union) + eps)
737    return iou

740def get_centers_and_bounding_boxes(
741    segmentation: np.ndarray,
742    mode: str = "v"
743) -> Tuple[Dict[int, np.ndarray], Dict[int, tuple]]:
744    """Returns the center coordinates of the foreground instances in the ground-truth.
745
746    Args:
747        segmentation: The segmentation.
748        mode: Determines the functionality used for computing the centers.
749        If 'v', the object's eccentricity centers computed by vigra are used.
750        If 'p' the object's centroids computed by skimage are used.
751
752    Returns:
753        A dictionary that maps object ids to the corresponding centroid.
754        A dictionary that maps object_ids to the corresponding bounding box.
755    """
756    assert mode in ["p", "v"], "Choose either 'p' for regionprops or 'v' for vigra"
757
758    properties = regionprops(segmentation)
759
760    if mode == "p":
761        center_coordinates = {prop.label: prop.centroid for prop in properties}
762    elif mode == "v":
763        center_coordinates = vigra.filters.eccentricityCenters(segmentation.astype('float32'))
764        center_coordinates = {i: coord for i, coord in enumerate(center_coordinates) if i > 0}
765
766    bbox_coordinates = {prop.label: prop.bbox for prop in properties}
767
768    assert len(bbox_coordinates) == len(center_coordinates), f"{len(bbox_coordinates)}, {len(center_coordinates)}"
769    return center_coordinates, bbox_coordinates

643
644        f = zarr.open(save_path, "a")

648            ("halo", halo if halo is None else list(halo)),
649            ("model_type", predictor.model_type)

725    """Compute the intersection over union of two masks.
726
727    Args:
728        mask1: The first mask.
729        mask2: The second mask.
730
731    Returns:
732        The intersection over union of the two masks.
733    """

735    union = np.logical_or(mask1 == 1, mask2 == 1).sum()
