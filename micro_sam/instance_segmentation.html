  44def mask_data_to_segmentation(
  45    masks: List[Dict[str, Any]],
  46    with_background: bool,
  47    min_object_size: int = 0,
  48    max_object_size: Optional[int] = None,
  49) -> np.ndarray:
  50    """Convert the output of the automatic mask generation to an instance segmentation.
  51
  52    Args:
  53        masks: The outputs generated by AutomaticMaskGenerator or EmbeddingMaskGenerator.
  54            Only supports output_mode=binary_mask.
  55        with_background: Whether the segmentation has background. If yes this function assures that the largest
  56            object in the output will be mapped to zero (the background value).
  57        min_object_size: The minimal size of an object in pixels.
  58        max_object_size: The maximal size of an object in pixels.
  59    Returns:
  60        The instance segmentation.
  61    """
  62
  63    masks = sorted(masks, key=(lambda x: x["area"]), reverse=True)
  64    # we could also get the shape from the crop box
  65    shape = next(iter(masks))["segmentation"].shape
  66    segmentation = np.zeros(shape, dtype="uint32")
  67
  68    def require_numpy(mask):
  69        return mask.cpu().numpy() if torch.is_tensor(mask) else mask
  70
  71    seg_id = 1
  72    for mask in masks:
  73        if mask["area"] < min_object_size:
  74            continue
  75        if max_object_size is not None and mask["area"] > max_object_size:
  76            continue
  77
  78        this_seg_id = mask.get("seg_id", seg_id)
  79        segmentation[require_numpy(mask["segmentation"])] = this_seg_id
  80        seg_id = this_seg_id + 1
  81
  82    seg_ids, sizes = np.unique(segmentation, return_counts=True)
  83
  84    # In some cases objects may be smaller than peviously calculated,
  85    # since they are covered by other objects. We ensure these also get
  86    # filtered out here.
  87    filter_ids = seg_ids[sizes < min_object_size]
  88
  89    # If we run segmentation with background we also map the largest segment
  90    # (the most likely background object) to zero. This is often zero already,
  91    # but it does not hurt to reset that to zero either.
  92    if with_background:
  93        bg_id = seg_ids[np.argmax(sizes)]
  94        filter_ids = np.concatenate([filter_ids, [bg_id]])
  95
  96    segmentation[np.isin(segmentation, filter_ids)] = 0
  97    vigra.analysis.relabelConsecutive(segmentation, out=segmentation)
  98
  99    return segmentation

 141    def _postprocess_batch(
 142        self,
 143        data,
 144        crop_box,
 145        original_size,
 146        pred_iou_thresh,
 147        stability_score_thresh,
 148        box_nms_thresh,
 149    ):
 150        orig_h, orig_w = original_size
 151
 152        # filter by predicted IoU
 153        if pred_iou_thresh > 0.0:
 154            keep_mask = data["iou_preds"] > pred_iou_thresh
 155            data.filter(keep_mask)
 156
 157        # filter by stability score
 158        if stability_score_thresh > 0.0:
 159            keep_mask = data["stability_score"] >= stability_score_thresh
 160            data.filter(keep_mask)
 161
 162        # filter boxes that touch crop boundaries
 163        keep_mask = ~amg_utils.is_box_near_crop_edge(data["boxes"], crop_box, [0, 0, orig_w, orig_h])
 164        if not torch.all(keep_mask):
 165            data.filter(keep_mask)
 166
 167        # remove duplicates within this crop.
 168        keep_by_nms = batched_nms(
 169            data["boxes"].float(),
 170            data["iou_preds"],
 171            torch.zeros_like(data["boxes"][:, 0]),  # categories
 172            iou_threshold=box_nms_thresh,
 173        )
 174        data.filter(keep_by_nms)
 175
 176        # return to the original image frame
 177        data["boxes"] = amg_utils.uncrop_boxes_xyxy(data["boxes"], crop_box)
 178        data["crop_boxes"] = torch.tensor([crop_box for _ in range(len(data["rles"]))])
 179        # the data from embedding based segmentation doesn't have the points
 180        # so we skip if the corresponding key can't be found
 181        try:
 182            data["points"] = amg_utils.uncrop_points(data["points"], crop_box)
 183        except KeyError:
 184            pass
 185
 186        return data

 188    def _postprocess_small_regions(self, mask_data, min_area, nms_thresh):
 189
 190        if len(mask_data["rles"]) == 0:
 191            return mask_data
 192
 193        # filter small disconnected regions and holes
 194        new_masks = []
 195        scores = []
 196        for rle in mask_data["rles"]:
 197            mask = amg_utils.rle_to_mask(rle)
 198
 199            mask, changed = amg_utils.remove_small_regions(mask, min_area, mode="holes")
 200            unchanged = not changed
 201            mask, changed = amg_utils.remove_small_regions(mask, min_area, mode="islands")
 202            unchanged = unchanged and not changed
 203
 204            new_masks.append(torch.as_tensor(mask, dtype=torch.int).unsqueeze(0))
 205            # give score=0 to changed masks and score=1 to unchanged masks
 206            # so NMS will prefer ones that didn't need postprocessing
 207            scores.append(float(unchanged))
 208
 209        # recalculate boxes and remove any new duplicates
 210        masks = torch.cat(new_masks, dim=0)
 211        boxes = batched_mask_to_box(masks)
 212        keep_by_nms = batched_nms(
 213            boxes.float(),
 214            torch.as_tensor(scores, dtype=torch.float),
 215            torch.zeros_like(boxes[:, 0]),  # categories
 216            iou_threshold=nms_thresh,
 217        )
 218
 219        # only recalculate RLEs for masks that have changed
 220        for i_mask in keep_by_nms:
 221            if scores[i_mask] == 0.0:
 222                mask_torch = masks[i_mask].unsqueeze(0)
 223                # mask_data["rles"][i_mask] = amg_utils.mask_to_rle_pytorch(mask_torch)[0]
 224                mask_data["rles"][i_mask] = mask_to_rle_pytorch(mask_torch)[0]
 225                mask_data["boxes"][i_mask] = boxes[i_mask]  # update res directly
 226        mask_data.filter(keep_by_nms)
 227
 228        return mask_data

 230    def _postprocess_masks(self, mask_data, min_mask_region_area, box_nms_thresh, crop_nms_thresh, output_mode):
 231        # filter small disconnected regions and holes in masks
 232        if min_mask_region_area > 0:
 233            mask_data = self._postprocess_small_regions(
 234                mask_data,
 235                min_mask_region_area,
 236                max(box_nms_thresh, crop_nms_thresh),
 237            )
 238
 239        # encode masks
 240        if output_mode == "coco_rle":
 241            mask_data["segmentations"] = [amg_utils.coco_encode_rle(rle) for rle in mask_data["rles"]]
 242        elif output_mode == "binary_mask":
 243            mask_data["segmentations"] = [amg_utils.rle_to_mask(rle) for rle in mask_data["rles"]]
 244        else:
 245            mask_data["segmentations"] = mask_data["rles"]
 246
 247        # write mask records
 248        curr_anns = []
 249        for idx in range(len(mask_data["segmentations"])):
 250            ann = {
 251                "segmentation": mask_data["segmentations"][idx],
 252                "area": amg_utils.area_from_rle(mask_data["rles"][idx]),
 253                "bbox": amg_utils.box_xyxy_to_xywh(mask_data["boxes"][idx]).tolist(),
 254                "predicted_iou": mask_data["iou_preds"][idx].item(),
 255                "stability_score": mask_data["stability_score"][idx].item(),
 256                "crop_box": amg_utils.box_xyxy_to_xywh(mask_data["crop_boxes"][idx]).tolist(),
 257            }
 258            # the data from embedding based segmentation doesn't have the points
 259            # so we skip if the corresponding key can't be found
 260            try:
 261                ann["point_coords"] = [mask_data["points"][idx].tolist()]
 262            except KeyError:
 263                pass
 264            curr_anns.append(ann)
 265
 266        return curr_anns

 268    def _to_mask_data(self, masks, iou_preds, crop_box, original_size, points=None):
 269        orig_h, orig_w = original_size
 270
 271        # serialize predictions and store in MaskData
 272        data = amg_utils.MaskData(masks=masks.flatten(0, 1), iou_preds=iou_preds.flatten(0, 1))
 273        if points is not None:
 274            data["points"] = torch.as_tensor(points.repeat(masks.shape[1], axis=0), dtype=torch.float)
 275
 276        del masks
 277
 278        # calculate the stability scores
 279        data["stability_score"] = amg_utils.calculate_stability_score(
 280            data["masks"], self._predictor.model.mask_threshold, self._stability_score_offset
 281        )
 282
 283        # threshold masks and calculate boxes
 284        data["masks"] = data["masks"] > self._predictor.model.mask_threshold
 285        data["masks"] = data["masks"].type(torch.bool)
 286        data["boxes"] = batched_mask_to_box(data["masks"])
 287
 288        # compress to RLE
 289        data["masks"] = amg_utils.uncrop_masks(data["masks"], crop_box, orig_h, orig_w)
 290        # data["rles"] = amg_utils.mask_to_rle_pytorch(data["masks"])
 291        data["rles"] = mask_to_rle_pytorch(data["masks"])
 292        del data["masks"]
 293
 294        return data

 327class AutomaticMaskGenerator(AMGBase):
 328    """Generates an instance segmentation without prompts, using a point grid.
 329
 330    This class implements the same logic as
 331    https://github.com/facebookresearch/segment-anything/blob/main/segment_anything/automatic_mask_generator.py
 332    It decouples the computationally expensive steps of generating masks from the cheap post-processing operation
 333    to filter these masks to enable grid search and interactively changing the post-processing.
 334
 335    Use this class as follows:
 336    ```python
 337    amg = AutomaticMaskGenerator(predictor)
 338    amg.initialize(image)  # Initialize the masks, this takes care of all expensive computations.
 339    masks = amg.generate(pred_iou_thresh=0.8)  # Generate the masks. This is fast and enables testing parameters
 340    ```
 341
 342    Args:
 343        predictor: The segment anything predictor.
 344        points_per_side: The number of points to be sampled along one side of the image.
 345            If None, `point_grids` must provide explicit point sampling.
 346        points_per_batch: The number of points run simultaneously by the model.
 347            Higher numbers may be faster but use more GPU memory.
 348        crop_n_layers: If >0, the mask prediction will be run again on crops of the image.
 349        crop_overlap_ratio: Sets the degree to which crops overlap.
 350        crop_n_points_downscale_factor: How the number of points is downsampled when predicting with crops.
 351        point_grids: A lisst over explicit grids of points used for sampling masks.
 352            Normalized to [0, 1] with respect to the image coordinate system.
 353        stability_score_offset: The amount to shift the cutoff when calculating the stability score.
 354    """
 355    def __init__(
 356        self,
 357        predictor: SamPredictor,
 358        points_per_side: Optional[int] = 32,
 359        points_per_batch: Optional[int] = None,
 360        crop_n_layers: int = 0,
 361        crop_overlap_ratio: float = 512 / 1500,
 362        crop_n_points_downscale_factor: int = 1,
 363        point_grids: Optional[List[np.ndarray]] = None,
 364        stability_score_offset: float = 1.0,
 365    ):
 366        super().__init__()
 367
 368        if points_per_side is not None:
 369            self.point_grids = amg_utils.build_all_layer_point_grids(
 370                points_per_side,
 371                crop_n_layers,
 372                crop_n_points_downscale_factor,
 373            )
 374        elif point_grids is not None:
 375            self.point_grids = point_grids
 376        else:
 377            raise ValueError("Can't have both points_per_side and point_grid be None or not None.")
 378
 379        self._predictor = predictor
 380        self._points_per_side = points_per_side
 381
 382        # we set the points per batch to 16 for mps for performance reasons
 383        # and otherwise keep them at the default of 64
 384        if points_per_batch is None:
 385            points_per_batch = 16 if str(predictor.device) == "mps" else 64
 386        self._points_per_batch = points_per_batch
 387
 388        self._crop_n_layers = crop_n_layers
 389        self._crop_overlap_ratio = crop_overlap_ratio
 390        self._crop_n_points_downscale_factor = crop_n_points_downscale_factor
 391        self._stability_score_offset = stability_score_offset
 392
 393    def _process_batch(self, points, im_size, crop_box, original_size):
 394        # run model on this batch
 395        transformed_points = self._predictor.transform.apply_coords(points, im_size)
 396        in_points = torch.as_tensor(transformed_points, device=self._predictor.device, dtype=torch.float)
 397        in_labels = torch.ones(in_points.shape[0], dtype=torch.int, device=in_points.device)
 398        masks, iou_preds, _ = self._predictor.predict_torch(
 399            in_points[:, None, :],
 400            in_labels[:, None],
 401            multimask_output=True,
 402            return_logits=True,
 403        )
 404        data = self._to_mask_data(masks, iou_preds, crop_box, original_size, points=points)
 405        del masks
 406        return data
 407
 408    def _process_crop(self, image, crop_box, crop_layer_idx, precomputed_embeddings, pbar_init=None, pbar_update=None):
 409        # Crop the image and calculate embeddings.
 410        x0, y0, x1, y1 = crop_box
 411        cropped_im = image[y0:y1, x0:x1, :]
 412        cropped_im_size = cropped_im.shape[:2]
 413
 414        if not precomputed_embeddings:
 415            self._predictor.set_image(cropped_im)
 416
 417        # Get the points for this crop.
 418        points_scale = np.array(cropped_im_size)[None, ::-1]
 419        points_for_image = self.point_grids[crop_layer_idx] * points_scale
 420
 421        # Generate masks for this crop in batches.
 422        data = amg_utils.MaskData()
 423        n_batches = len(points_for_image) // self._points_per_batch +\
 424            int(len(points_for_image) % self._points_per_batch != 0)
 425        if pbar_init is not None:
 426            pbar_init(n_batches, "Predict masks for point grid prompts")
 427
 428        for (points,) in amg_utils.batch_iterator(self._points_per_batch, points_for_image):
 429            batch_data = self._process_batch(points, cropped_im_size, crop_box, self.original_size)
 430            data.cat(batch_data)
 431            del batch_data
 432            if pbar_update is not None:
 433                pbar_update(1)
 434
 435        if not precomputed_embeddings:
 436            self._predictor.reset_image()
 437
 438        return data

 440    @torch.no_grad()
 441    def initialize(
 442        self,
 443        image: np.ndarray,
 444        image_embeddings: Optional[util.ImageEmbeddings] = None,
 445        i: Optional[int] = None,
 446        verbose: bool = False,
 447        pbar_init: Optional[callable] = None,
 448        pbar_update: Optional[callable] = None,
 449    ) -> None:
 450        """Initialize image embeddings and masks for an image.
 451
 452        Args:
 453            image: The input image, volume or timeseries.
 454            image_embeddings: Optional precomputed image embeddings.
 455                See `util.precompute_image_embeddings` for details.
 456            i: Index for the image data. Required if `image` has three spatial dimensions
 457                or a time dimension and two spatial dimensions.
 458            verbose: Whether to print computation progress.
 459            pbar_init: Callback to initialize an external progress bar. Must accept number of steps and description.
 460                Can be used together with pbar_update to handle napari progress bar in other thread.
 461                To enables using this function within a threadworker.
 462            pbar_update: Callback to update an external progress bar.
 463        """
 464        original_size = image.shape[:2]
 465        self._original_size = original_size
 466
 467        crop_boxes, layer_idxs = amg_utils.generate_crop_boxes(
 468            original_size, self._crop_n_layers, self._crop_overlap_ratio
 469        )
 470
 471        # We can set fixed image embeddings if we only have a single crop box (the default setting).
 472        # Otherwise we have to recompute the embeddings for each crop and can't precompute.
 473        if len(crop_boxes) == 1:
 474            if image_embeddings is None:
 475                image_embeddings = util.precompute_image_embeddings(self._predictor, image)
 476            util.set_precomputed(self._predictor, image_embeddings, i=i)
 477            precomputed_embeddings = True
 478        else:
 479            precomputed_embeddings = False
 480
 481        # we need to cast to the image representation that is compatible with SAM
 482        image = util._to_image(image)
 483
 484        _, pbar_init, pbar_update, pbar_close = util.handle_pbar(verbose, pbar_init, pbar_update)
 485
 486        crop_list = []
 487        for crop_box, layer_idx in zip(crop_boxes, layer_idxs):
 488            crop_data = self._process_crop(
 489                image, crop_box, layer_idx,
 490                precomputed_embeddings=precomputed_embeddings,
 491                pbar_init=pbar_init, pbar_update=pbar_update,
 492            )
 493            crop_list.append(crop_data)
 494        pbar_close()
 495
 496        self._is_initialized = True
 497        self._crop_list = crop_list
 498        self._crop_boxes = crop_boxes

 500    @torch.no_grad()
 501    def generate(
 502        self,
 503        pred_iou_thresh: float = 0.88,
 504        stability_score_thresh: float = 0.95,
 505        box_nms_thresh: float = 0.7,
 506        crop_nms_thresh: float = 0.7,
 507        min_mask_region_area: int = 0,
 508        output_mode: str = "binary_mask",
 509    ) -> List[Dict[str, Any]]:
 510        """Generate instance segmentation for the currently initialized image.
 511
 512        Args:
 513            pred_iou_thresh: Filter threshold in [0, 1], using the mask quality predicted by the model.
 514            stability_score_thresh: Filter threshold in [0, 1], using the stability of the mask
 515                under changes to the cutoff used to binarize the model prediction.
 516            box_nms_thresh: The IoU threshold used by nonmax suppression to filter duplicate masks.
 517            crop_nms_thresh: The IoU threshold used by nonmax suppression to filter duplicate masks between crops.
 518            min_mask_region_area: Minimal size for the predicted masks.
 519            output_mode: The form masks are returned in.
 520
 521        Returns:
 522            The instance segmentation masks.
 523        """
 524        if not self.is_initialized:
 525            raise RuntimeError("AutomaticMaskGenerator has not been initialized. Call initialize first.")
 526
 527        data = amg_utils.MaskData()
 528        for data_, crop_box in zip(self.crop_list, self.crop_boxes):
 529            crop_data = self._postprocess_batch(
 530                data=deepcopy(data_),
 531                crop_box=crop_box, original_size=self.original_size,
 532                pred_iou_thresh=pred_iou_thresh,
 533                stability_score_thresh=stability_score_thresh,
 534                box_nms_thresh=box_nms_thresh
 535            )
 536            data.cat(crop_data)
 537
 538        if len(self.crop_boxes) > 1 and len(data["crop_boxes"]) > 0:
 539            # Prefer masks from smaller crops
 540            scores = 1 / box_area(data["crop_boxes"])
 541            scores = scores.to(data["boxes"].device)
 542            keep_by_nms = batched_nms(
 543                data["boxes"].float(),
 544                scores,
 545                torch.zeros_like(data["boxes"][:, 0]),  # categories
 546                iou_threshold=crop_nms_thresh,
 547            )
 548            data.filter(keep_by_nms)
 549
 550        data.to_numpy()
 551        masks = self._postprocess_masks(data, min_mask_region_area, box_nms_thresh, crop_nms_thresh, output_mode)
 552        return masks

 556def _process_tiled_embeddings(predictor, image, image_embeddings, tile_shape, halo):
 557    if image_embeddings is None:
 558        if tile_shape is None or halo is None:
 559            raise ValueError("To compute tiled embeddings the parameters tile_shape and halo have to be passed.")
 560        image_embeddings = util.precompute_image_embeddings(predictor, image, tile_shape=tile_shape, halo=halo)

 693class DecoderAdapter(torch.nn.Module):
 694    """Adapter to contain the UNETR decoder in a single module.
 695
 696    To apply the decoder on top of pre-computed embeddings for
 697    the segmentation functionality.
 698    See also: https://github.com/constantinpape/torch-em/blob/main/torch_em/model/unetr.py
 699    """
 700    def __init__(self, unetr):
 701        super().__init__()
 702
 703        self.base = unetr.base
 704        self.out_conv = unetr.out_conv
 705        self.deconv_out = unetr.deconv_out
 706        self.decoder_head = unetr.decoder_head
 707        self.final_activation = unetr.final_activation
 708        self.postprocess_masks = unetr.postprocess_masks
 709
 710        self.decoder = unetr.decoder
 711        self.deconv1 = unetr.deconv1
 712        self.deconv2 = unetr.deconv2
 713        self.deconv3 = unetr.deconv3
 714        self.deconv4 = unetr.deconv4
 715
 716    def forward(self, input_, input_shape, original_shape):
 717        z12 = input_
 718
 719        z9 = self.deconv1(z12)
 720        z6 = self.deconv2(z9)
 721        z3 = self.deconv3(z6)
 722        z0 = self.deconv4(z3)
 723
 724        updated_from_encoder = [z9, z6, z3]
 725
 726        x = self.base(z12)
 727        x = self.decoder(x, encoder_inputs=updated_from_encoder)
 728        x = self.deconv_out(x)
 729
 730        x = torch.cat([x, z0], dim=1)
 731        x = self.decoder_head(x)
 732
 733        x = self.out_conv(x)
 734        if self.final_activation is not None:
 735            x = self.final_activation(x)
 736
 737        x = self.postprocess_masks(x, input_shape, original_shape)
 738        return x

 837    ```

 839    Args:
 840        predictor: The segment anything predictor.

 842            for instance segmentation.
 843    """
 844    def __init__(
 845        self,
 846        predictor: SamPredictor,
 847        decoder: torch.nn.Module,
 848    ) -> None:
 849        self._predictor = predictor
 850        self._decoder = decoder
 851
 852        # The decoder outputs.
 853        self._foreground = None

 855        self._boundary_distances = None

 867        self,

 873        pbar_update: Optional[callable] = None,

 876
 877        Args:
 878            image: The input image, volume or timeseries.

 880                See `util.precompute_image_embeddings` for details.

 891

 893            image_embeddings = util.precompute_image_embeddings(self._predictor, image)
 894

 897        embeddings = self._predictor.features
 898        input_shape = tuple(self._predictor.input_size)
 899        original_shape = tuple(self._predictor.original_size)
 900

 902        output = self._decoder(embeddings, input_shape, original_shape).cpu().numpy().squeeze(0)
 903        assert output.shape[0] == 3, f"{output.shape}"

 913    def _to_masks(self, segmentation, output_mode):
 914        if output_mode != "binary_mask":
 915            raise NotImplementedError
 916
 917        props = regionprops(segmentation)
 918        ndim = segmentation.ndim
 919        assert ndim in (2, 3)
 920
 921        shape = segmentation.shape
 922        if ndim == 2:
 923            crop_box = [0, shape[1], 0, shape[0]]
 924        else:
 925            crop_box = [0, shape[2], 0, shape[1], 0, shape[0]]
 926
 927        # go from skimage bbox in format [y0, x0, y1, x1] to SAM format [x0, w, y0, h]
 928        def to_bbox_2d(bbox):
 929            y0, x0 = bbox[0], bbox[1]
 930            w = bbox[3] - x0
 931            h = bbox[2] - y0
 932            return [x0, w, y0, h]
 933
 934        def to_bbox_3d(bbox):
 935            z0, y0, x0 = bbox[0], bbox[1], bbox[2]
 936            w = bbox[5] - x0
 937            h = bbox[4] - y0
 938            d = bbox[3] - y0
 939            return [x0, w, y0, h, z0, d]
 940
 941        to_bbox = to_bbox_2d if ndim == 2 else to_bbox_3d
 942        masks = [
 943            {
 944                "segmentation": segmentation == prop.label,
 945                "area": prop.area,
 946                "bbox": to_bbox(prop.bbox),
 947                "crop_box": crop_box,
 948                "seg_id": prop.label,
 949            } for prop in props
 950        ]
 951        return masks

 955        self,
 956        center_distance_threshold: float = 0.5,
 957        boundary_distance_threshold: float = 0.5,

 964        """Generate instance segmentation for the currently initialized image.
 965
 966        Args:

 981        if not self.is_initialized:
 982            raise RuntimeError("InstanceSegmentationWithDecoder has not been initialized. Call initialize first.")
 983
 984        if foreground_smoothing > 0:
 985            foreground = vigra.filters.gaussianSmoothing(self._foreground, foreground_smoothing)

 993            boundary_distance_threshold=boundary_distance_threshold,
 994            foreground_threshold=foreground_threshold,

 64    masks = sorted(masks, key=(lambda x: x["area"]), reverse=True)
 65    # we could also get the shape from the crop box
 66    shape = next(iter(masks))["segmentation"].shape
 67    segmentation = np.zeros(shape, dtype="uint32")
 68
 69    def require_numpy(mask):
 70        return mask.cpu().numpy() if torch.is_tensor(mask) else mask
 71
 72    seg_id = 1
 73    for mask in masks:
 74        if mask["area"] < min_object_size:
 75            continue
 76        if max_object_size is not None and mask["area"] > max_object_size:
 77            continue
 78
 79        this_seg_id = mask.get("seg_id", seg_id)
 80        segmentation[require_numpy(mask["segmentation"])] = this_seg_id
 81        seg_id = this_seg_id + 1
 82
 83    seg_ids, sizes = np.unique(segmentation, return_counts=True)
 84
 85    # In some cases objects may be smaller than peviously calculated,
 86    # since they are covered by other objects. We ensure these also get
 87    # filtered out here.
 88    filter_ids = seg_ids[sizes < min_object_size]
 89
 90    # If we run segmentation with background we also map the largest segment
 91    # (the most likely background object) to zero. This is often zero already,
 92    # but it does not hurt to reset that to zero either.
 93    if with_background:
 94        bg_id = seg_ids[np.argmax(sizes)]
 95        filter_ids = np.concatenate([filter_ids, [bg_id]])
 96
 97    segmentation[np.isin(segmentation, filter_ids)] = 0
 98    vigra.analysis.relabelConsecutive(segmentation, out=segmentation)
 99
100    return segmentation
